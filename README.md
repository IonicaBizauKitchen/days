## Usage

```js
var days = require('days');

console.log(days);
//=> ['Sunday', 'Monday', 'Tuesday', 'Wendesday', 'Thursday', 'Friday', 'Saturday']

console.log(days.abbr)
//=> ['Sun', 'Mon', 'Tue', 'Wed', 'Thur', 'Fri', 'Sat']

console.log(days.short)
//=> ['Su', 'Mo', 'Tu', 'We', 'Th', 'Fr', 'Sa']
```
